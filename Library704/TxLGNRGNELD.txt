.Module TxLGNRGNELD
LH_GridNeg RH_GridNegEdge Reset LH_Divider_Junction : TxLGNRGNELD
.Signals
I LH_GridNeg 
I RH_GridNegEdge 
I Reset 
O LH_Divider_Junction
.Logic
reg State=1;

always @*
  if(Reset)
     State=1;
  else if(~LH_GridNeg)
     State=0;
   
always @(negedge RH_GridNegEdge)
   if(~Reset&LH_GridNeg)
      State<=1;

assign LH_Divider_Junction=~State;
.End
