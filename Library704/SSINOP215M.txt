.Module SSINOP215M // Single shot
InputNeg OutputPos Clock32768 Res : SSINOP215M
.Signals
I InputNeg
O OutputPos
I Clock32768
I Res
.Logic

reg State=0;
reg[2:0] Cnt=0;

always @(posedge Clock32768)
begin
if(State)
begin
  Cnt++;
  if(Cnt>6)
     State=0;
end
end

always @(negedge InputNeg)
if(~Res)
  begin
    Cnt=0;
    State=1;  
  end


assign OutputPos=State;
.End
