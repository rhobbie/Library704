.Module SSINOP215M // Single shot
InputNeg OutputPos Clock32768 : SSINOP215M
.Signals
I InputNeg
O OutputPos
I Clock32768

.Logic

wire State;
wire[2:0] Cnt;

initial State=0;

always @(posedge Clock32768)
begin
if(State)
begin
  Cnt++;
  if(Cnt>=6)
     State=0;
end
end

always @(posedge InputNeg)
begin
  Cnt=0;
  State=1;  
end

assign OutputPos=State;
.End
