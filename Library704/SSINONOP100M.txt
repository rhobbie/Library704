.Module SSINONOP100M // Single shot , negative and positive pulse
InputNeg OutputNeg OutputPos  Clock16384 : SSINONOP100M
.Signals
I InputNeg
O OutputNeg
O OutputPos
I Clock16384

.Logic
wire State;
wire[2:0] Cnt;

initial State=0;

always @(posedge Clock16384)
begin
if(State)
begin
  Cnt++;
  if(Cnt>=6)
     State=0;
end
end

always @(negedge InputNeg)
begin
  Cnt=0;
  State=1;  
end

assign OutputPos=State;
assign OutputNeg=!State;
.End
