.Module CE	
CE[01-04][a-e] : CE
MACH_CYCLE_HAND_KEY_OP MACH_CYCLE_HAND_KEY_NO : BUTTON
AUX_RESET_OP AUX_RESET_NO : BUTTON_B
AUX_START_OP AUX_START_NO : BUTTON_B
CONTINUOUS_STORAGE_READ_IN_OP CONTINUOUS_STORAGE_READ_IN_NO : BUTTON

.Signals
I CE04a +150 ON TRUE MANUAL
T CE03a +150 ON TRUE MANUAL
O CE04b MACH CYCLE HAND KEY
I CE04e AUX RESET OP
B CE04c AUX RESET NO
B CE04d AUX START NO
I CE02b CONTINUOUS STORAGE READ IN OP
O CE02c CONTINUOUS STORAGE READ IN NO

.Connect
// 4.05.01 TURN OFF MSTR STOP TGR
// AUX_START_OP AUX_START_NO : BUTTON_B
W +40V AUX_START_OP 
W AUX_START_NO CE04d // AUX START NO

// 4.07 MACH CYCLE, SINGLE & MULT STEP
W CE04a CE03a // +150 ON TRUE MANUAL
// MACH_CYCLE_HAND_KEY_OP MACH_CYCLE_HAND_KEY_NO : BUTTON
W CE03a MACH_CYCLE_HAND_KEY_OP // +150 ON TRUE MANUAL
220C MACH_CYCLE_HAND_KEY_NO 0V 
200k MACH_CYCLE_HAND_KEY_NO 0V 
W MACH_CYCLE_HAND_KEY_NO CE04b // MACH CYCLE HAND KEY

// 4.09 RESET CKTS #2
// AUX_RESET_OP AUX_RESET_NO : BUTTON_B
W CE04e  AUX_RESET_OP // AUX RESET OP
W AUX_RESET_NO CE04c // AUX RESET NO

// 4.17 ENTER INSTRUCTION TIMER
// CONTINUOUS_STORAGE_READ_IN_OP CONTINUOUS_STORAGE_READ_IN_NO : BUTTON
W CE02b CONTINUOUS_STORAGE_READ_IN_OP // CONTINUOUS STORAGE READ IN OP
W CONTINUOUS_STORAGE_READ_IN_NO CE02c // CONTINUOUS STORAGE READ IN NO

.End
