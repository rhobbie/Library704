.Module SSIPOP1M // Single shot 1ms duration, 128µs clock input
InputPos OutputPos Clock128 Res : SSIPOP1M
.Signals
I InputPos
O OutputPos
I Clock128 
I Res
.Logic

reg State;
reg[2:0] Cnt;

initial State=0;

always @(posedge Clock128)
begin
if(State)
begin
  Cnt++;
  if(Cnt>=7)
     State=0;
end
end

always @(posedge InputPos)
  if(!Res)
  begin
    Cnt=0;
    State=1;  
  end


assign OutputPos=State;
.End
