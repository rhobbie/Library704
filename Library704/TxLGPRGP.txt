.Module TxLGPRGP	
LH_GridPos RH_GridPos Reset LH_Divider_Junction RH_Divider_Junction : TxLGPRGP
.Signals
I LH_GridPos
I RH_GridPos
I Reset
O LH_Divider_Junction
O RH_Divider_Junction
.Logic

   if(LH_GridPos|Reset)
        State=1;
