.Module SSIPONOP500 // Single shot , negative and positive pulse
InputPos OutputNeg OutputPos  Clock64 : SSIPONOP500
.Signals
I InputPos
O OutputNeg
O OutputPos
I Clock64
.Logic

reg State;
reg[2:0] Cnt;

initial State=0;

always @(posedge Clock64)
begin
if(State)
begin
  Cnt++;
  if(Cnt>=7)
     State=0;
end
end

always @(posedge InputPos)
begin
  Cnt=0;
  State=1;  
end

assign OutputPos=State;
.End
