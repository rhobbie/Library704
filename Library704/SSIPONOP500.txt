.Module SSIPONOP500 // Single shot , negative and positive pulse
InputPos OutputNeg OutputPos  Clock64 Res : SSIPONOP500
.Signals
I InputPos
O OutputNeg
O OutputPos
I Clock64
I Res
.Logic

reg State;
reg[3:0] Cnt;

initial State=0;

always @(posedge Clock64)
begin
if(State)
begin
  Cnt++;
  if(Cnt>=8)
     State=0;
end
end

always @(posedge InputPos)
  if(!Res)
    begin
    Cnt=0;
  State=1;  
  end


assign OutputPos=State;
assign OutputNeg=!State;
.End
