.Module D	// Microsecond Delay Unit  B101
Input Output Clock Reset : D		
.Signals
I Input
O Output
I Clock
I Reset
.Logic
reg State=0;
reg preState=0;


always @*
   if(Reset)
   begin
      State=0;
	  preState=0;
   end

always @(negedge Clock) 
    if(~Reset)
	    preState<=Input;
always @(posedge Clock)
    if(~Reset)
        State<=preState;
    
assign Output=State;

.End
