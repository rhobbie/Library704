.Module TLGPBRGPRDx
LH_GridPos Binary RH_GridPos Reset RH_Divider_Junction : TLGPBRGPRDx
.Signals
I LH_GridPos
I Binary
I RH_GridPos
I Reset
O RH_Divider_Junction
.Logic

reg State=0;

always @*
begin
    if(Reset|RH_GridPos)
        State=0;    
    else if(LH_GridPos) 
        State=1;
end

always @(negedge Binary) 
  if(~Reset&~RH_GridPos&~LH_GridPos)
      State<=~State;

assign RH_Divider_Junction=State;

.End
