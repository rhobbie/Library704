.Module TLGPBRGPRDx
LH_GridPos Binary RH_GridPos Reset RH_Divider_Junction : TLGPBRGPRDx
.Signals
I LH_GridPos
I Binary
I RH_GridPos
I Reset
O RH_Divider_Junction
.Logic

reg State;

initial State=0;

always @(Reset, LH_GridPos, RH_GridPos)
begin
    if(Reset||RH_GridPos)
        assign State=0;    
    else if(LH_GridPos) 
        assign State=1;
    else
        deassign State;
end

always @(negedge Binary) 
    State=!State;

assign RH_Divider_Junction=State;

.End
