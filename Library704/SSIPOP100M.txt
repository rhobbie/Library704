.Module SSIPOP100M // Single shot
InputPos OutputPos Clock16384 Res : SSIPOP100M
.Signals
I InputPos
O OutputPos
I Clock16384
I Res
.Logic
reg State;
reg[2:0] Cnt;

initial State=0;

always @(posedge Clock16384)
begin
if(State)
begin
  Cnt++;
  if(Cnt>6)
     State=0;
end
end

always @(posedge InputPos)
  if(!Res)
  begin
    Cnt=0;
    State=1;  
  end


assign OutputPos=State;
.End
