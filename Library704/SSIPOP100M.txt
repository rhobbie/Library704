.Module SSIPOP100M // Single shot
InputPos OutputPos Clock16384 Res : SSIPOP100M
.Signals
I InputPos
O OutputPos
I Clock16384
I Res
.Logic
reg State=0;
reg[2:0] Cnt=0;


always @(posedge Clock16384)
begin
if(State)
begin
  Cnt++;
  if(Cnt==7)
     State=0;
end
end

always @(posedge InputPos)
  if(~Res)
  begin
    Cnt=0;
    State=1;  
  end


assign OutputPos=State;
.End
