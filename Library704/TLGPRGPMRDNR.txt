.Module TLGPRGPMRDNR
LH_GridPos RH_GridPos Manual_On Manual_Off RH_Divider_Junction : TLGPRGPMRDNR
.Signals
I LH_GridPos
I RH_GridPos
I Manual_On
I Manual_Off
O RH_Divider_Junction
.Logic
reg State;

initial State=0;

always @(LH_GridPos or RH_GridPos)
begin
    if(RH_GridPos)
        State=0;    
    else if(LH_GridPos) 
        State=1;
end
assign RH_Divider_Junction=State;
.End
