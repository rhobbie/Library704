.Module TLGNBRDx	
LH_GridNeg Binary Reset RH_Divider_Junction : TLGNBRDx
.Signals
I LH_GridNeg
I Binary
I Reset
O RH_Divider_Junction
.Logic

reg State=0;

always @*
    if(~LH_GridNeg|Reset) 
        State=0;

always @(negedge Binary) 
    if(LH_GridNeg&~Reset) 
		State<=~State;

assign RH_Divider_Junction=State;

.End
