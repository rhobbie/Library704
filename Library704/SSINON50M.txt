.Module  SSINON50M // Single shot Negative pulse
InputNeg OutputNeg Clock8192 : SSINON50M
.Signals
I InputNeg
O OutputNeg
I Clock8192

.Logic
wire State;
wire[2:0] Cnt;

initial State=0;

always @(posedge Clock8192)
begin
if(State)
{
  Cnt++;
  if(Cnt>=6)
  {
     State=0;
  }
}
end

always @(negedge InputNeg)
begin
  Cnt=0;
  State=1;  
end

assign OutputNeg=!State;
.End
