.Module  SSINON50M // Single shot Negative pulse
InputNeg OutputNeg Clock8192 Res : SSINON50M
.Signals
I InputNeg
O OutputNeg
I Clock8192
I Res

.Logic
reg State;
reg[2:0] Cnt;

initial State=0;

always @(posedge Clock8192)
begin
if(State)
begin
  Cnt++;
  if(Cnt>=6)
     State=0;
end
end

always @(negedge InputNeg)
  if(!Res)
  begin
    Cnt=0;
    State=1;  
  end

assign OutputNeg=!State;
.End
