.Module SSIPOP200M // Single shot
InputPos OutputPos Clock32768 Res : SSIPOP200M
.Signals
I InputPos
O OutputPos
I Clock32768
I Res
.Logic

reg State=0;
reg[2:0] Cnt=0;


always @(posedge Clock32768)
begin
if(State)
begin
  Cnt++;
  if(Cnt==6)
     State=0;
end
end

always @(posedge InputPos)
  if(~Res)
  begin
    Cnt=0;
    State=1;  
  end


assign OutputPos=State;
.End
