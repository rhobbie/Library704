.Module TxLGNRGNERD
LH_GridNeg RH_GridNegEdge Reset RH_Divider_Junction : TxLGNRGNERD
.Signals
I LH_GridNeg 
I RH_GridNegEdge 
I Reset 
O RH_Divider_Junction
.Logic
reg State=1;

always @*
if(Reset)
   State=1;
else if(~LH_GridNeg)
   State=0;

always @(negedge RH_GridNegEdge)
  if(~Reset&LH_GridNeg)
     State<=1;

assign RH_Divider_Junction=State;
.End
