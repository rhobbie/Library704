.Module TxLGNELGPMRD
LH_GridNegEdge LH_GridPos  Manual_On Manual_Off  Reset RH_Divider_Junction : TxLGNELGPMRD
.Signals
I LH_GridNegEdge
I LH_GridPos
M Manual_On
M Manual_Off
I Reset
O RH_Divider_Junction
.Logic
reg State=1;


always @*
  if(LH_GridPos|Reset)
     State=1;

always @(negedge LH_GridNegEdge) 
   if(~LH_GridPos&~Reset)
     State<=0;
   
assign RH_Divider_Junction=State;
.End
