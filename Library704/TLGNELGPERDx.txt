.Module TLGNELGPERDx	
LH_GridNegEdge LH_GridPosEdge Reset RH_Divider_Junction : TLGNELGPERDx

.Signals
I LH_GridNegEdge
I LH_GridPosEdge
I Reset
O RH_Divider_Junction
.Logic
reg State;

initial State=0;

always @(Reset)
if(Reset)
  assign State=0;
else
  deassign State;

always @(negedge LH_GridNegEdge) 
   State=0;
   
always @(posedge LH_GridPosEdge)
   State=1;

assign RH_Divider_Junction=State;

.End
