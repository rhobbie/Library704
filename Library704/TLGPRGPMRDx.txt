.Module TLGPRGPMRDx
LH_GridPos RH_GridPos Manual_On Manual_Off Reset RH_Divider_Junction : TLGPRGPMRDx
.Signals
I LH_GridPos
I RH_GridPos
I Manual_On
I Manual_Off
I Reset
O RH_Divider_Junction
.Logic
reg State;

initial State=0;

always @(LH_GridPos, RH_GridPos, Reset)
begin
    if(RH_GridPos||Reset)
        State=0;    
    else if(LH_GridPos) 
        State=1;
end
assign RH_Divider_Junction=State;
.End
