.Module SSINOP750 // Single shot 50u duration, 64µs clock input
InputNeg OutputPos Clock64 Res : SSINOP750

.Signals
I InputNeg
O OutputPos
I Clock64
I Res
.Logic
reg State=0;
reg[3:0] Cnt=0;


always @(posedge Clock64)
begin
if(State)
begin
  Cnt++;
  if(Cnt==12)
     State=0;
end
end

always @(negedge InputNeg)
  if(~Res)
  begin
    Cnt=0;
    State=1;  
  end

assign OutputPos=State;

.End
